    int primeNumberCount, number, i, j;

    Присваивание переменным значений
    primeNumberCount := 10;
    number := 0;

    while (primeNumberCount > 0)
    begin
    Рассматриваем следующее число
    number := number + 1;
    Количество делителей числа
    j := 0;

    Перебор всех чисел от 1 до number
    for i := 1 to number step 1
    begin

    Если делится без остатка на перебираемое число, то увеличить кол-во делителей
    if (number / i * i == number) j := j + 1

    end
    next;

    Если делителей только 2, то мы нашли простое число, делаем декремент primeNumberCount
    if (j == 2) primeNumberCount := primeNumberCount - 1

    end;

    Выводим последнее найденное число
    writeln number;
}
